    39вҶ’    const [total, list] = await Promise.all([
    40вҶ’      this.prisma.mistake.count({ where }),
    41вҶ’      this.prisma.mistake.findMany({
    42вҶ’        where,
    43вҶ’        include: {
    44вҶ’          subject: {
    45вҶ’            select: { id: true, name: true, icon: true, color: true },
    46вҶ’          },
    47вҶ’        },
    48вҶ’        orderBy: { createdAt: 'desc' },
    49вҶ’        skip: (page - 1) * size,
    50вҶ’        take: size,
    51вҶ’      }),
    52вҶ’    ]);
    53вҶ’
    54вҶ’    return {
    55вҶ’      total,

   237вҶ’  private calculateNextReview(reviewCount: number, result: string): Date {
   238вҶ’    const intervals = [1, 2, 4, 7, 15, 30, 60, 90]; // еӨ©ж•°
   239вҶ’
   240вҶ’    let idx = result === 'wrong' ? 0 : Math.min(reviewCount, intervals.length - 1);

   244вҶ’
