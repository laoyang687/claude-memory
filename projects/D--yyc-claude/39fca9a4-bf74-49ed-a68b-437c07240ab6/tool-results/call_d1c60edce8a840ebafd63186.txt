     1→import { Injectable } from '@nestjs/common';
     2→import { PrismaService } from '@/common/services/prisma.service';
     3→
     4→@Injectable()
     5→export class StatsService {
     6→  constructor(private prisma: PrismaService) {}
     7→
     8→  /**
     9→   * 获取学习概览
    10→   */
    11→  async getOverview(userId: number) {
    12→    const [totalMistakes, masteredCount, todayReviewed, todayAdded, subjectStats] =
    13→      await Promise.all([
    14→        this.prisma.mistake.count({ where: { userId: BigInt(userId) } }),
    15→        this.prisma.mistake.count({
    16→          where: { userId: BigInt(userId), mastered: true },
    17→        }),
    18→        this.getTodayReviewedCount(userId),
    19→        this.getTodayAddedCount(userId),
    20→        this.getSubjectStats(userId),
    21→      ]);
    22→
    23→    const masterRate = totalMistakes > 0
    24→      ? Number(((masteredCount / totalMistakes) * 100).toFixed(1))
    25→      : 0;
    26→
    27→    // 计算连续复习天数
    28→    const reviewStreak = await this.calculateReviewStreak(userId);
    29→
    30→    return {
    31→      totalMistakes,
    32→      masteredCount,
    33→      masterRate,
    34→      reviewStreak,
    35→      todayReviewed,
    36→      todayAdded,
    37→      subjectStats,
    38→    };
    39→  }
    40→
    41→  /**
    42→   * 获取科目分析
    43→   */
    44→  async getSubjectAnalysis(userId: number, subjectId: number) {
    45→    const [totalMistakes, masteredCount, weakKnowledge] = await Promise.all([
    46→      this.prisma.mistake.count({
    47→        where: { userId: BigInt(userId), subjectId },
    48→      }),
    49→      this.prisma.mistake.count({
    50→        where: { userId: BigInt(userId), subjectId, mastered: true },
    51→      }),
    52→      this.getWeakKnowledgePoints(userId, subjectId),
    53→    ]);
    54→
    55→    const masterRate = totalMistakes > 0
    56→      ? Number(((masteredCount / totalMistakes) * 100).toFixed(1))
    57→      : 0;
    58→
    59→    // 获取趋势数据（最近30天）
    60→    const trend = await this.getMasteryTrend(userId, subjectId, 30);
    61→
    62→    return {
    63→      subjectId,
    64→      totalMistakes,
    65→      masterRate,
    66→      weakKnowledge,
    67→      trend,
    68→    };
    69→  }
    70→
    71→  /**
    72→   * 获取今日复习数量
    73→   */
    74→  private async getTodayReviewedCount(userId: number): Promise<number> {
    75→    const today = new Date();
    76→    today.setHours(0, 0, 0, 0);
    77→    const tomorrow = new Date(today);
    78→    tomorrow.setDate(tomorrow.getDate() + 1);
    79→
    80→    return this.prisma.reviewRecord.count({
    81→      where: {
    82→        userId: BigInt(userId),
    83→        createdAt: { gte: today, lt: tomorrow },
    84→      },
    85→    });
    86→  }
    87→
    88→  /**
    89→   * 获取今日添加错题数量
    90→   */
    91→  private async getTodayAddedCount(userId: number): Promise<number> {
    92→    const today = new Date();
    93→    today.setHours(0, 0, 0, 0);
    94→    const tomorrow = new Date(today);
    95→    tomorrow.setDate(tomorrow.getDate() + 1);
    96→
    97→    return this.prisma.mistake.count({
    98→      where: {
    99→        userId: BigInt(userId),
   100→        createdAt: { gte: today, lt: tomorrow },
   101→      },
   102→    });
   103→  }
   104→
   105→  /**
   106→   * 获取各科目统计
   107→   */
   108→  private async getSubjectStats(userId: number) {
   109→    const result = await this.prisma.mistake.groupBy({
   110→      by: ['subjectId'],
   111→      where: { userId: BigInt(userId) },
   112→      _count: { subjectId: true },
   113→      _avg: { difficulty: true },
   114→    });
   115→
   116→    const subjects = await this.prisma.subject.findMany();
   117→
   118→    return result.map(r => {
   119→      const subject = subjects.find(s => s.id === r.subjectId);
   120→      return {
   121→        subject: subject ? { id: subject.id, name: subject.name, icon: subject.icon, color: subject.color } : null,
   122→        total: r._count.subjectId,
   123→        avgDifficulty: r._avg.difficulty ? Number(r._avg.difficulty.toFixed(1)) : 3,
   124→      };
   125→    });
   126→  }
   127→
   128→  /**
   129→   * 获取薄弱知识点
   130→   */
   131→  private async getWeakKnowledgePoints(userId: number, subjectId: number) {
   132→    // 获取该科目下所有错题的知识点分布
   133→    const mistakes = await this.prisma.mistake.findMany({
   134→      where: { userId: BigInt(userId), subjectId, mastered: false },
   135→      select: { knowledgeIds: true },
   136→    });
   137→
   138→    const knowledgeCount: Record<string, number> = {};
   139→
   140→    mistakes.forEach(m => {
   141→      if (m.knowledgeIds) {
   142→        m.knowledgeIds.split(',').forEach(id => {
   143→          knowledgeCount[id] = (knowledgeCount[id] || 0) + 1;
   144→        });
   145→      }
   146→    });
   147→
   148→    // 获取知识点详情
   149→    const knowledgeIds = Object.keys(knowledgeCount).map(id => parseInt(id));
   150→    const points = await this.prisma.knowledgePoint.findMany({
   151→      where: { id: { in: knowledgeIds } },
   152→    });
   153→
   154→    return points
   155→      .map(p => ({
   156→        id: p.id,
   157→        name: p.name,
   158→        mistakeCount: knowledgeCount[String(p.id)],
   159→      }))
   160→      .sort((a, b) => b.mistakeCount - a.mistakeCount)
   161→      .slice(0, 10);
   162→  }
   163→
   164→  /**
   165→   * 获取掌握率趋势
   166→   */
   167→  private async getMasteryTrend(userId: number, subjectId: number, days: number) {
   168→    const trend = [];
   169→    const now = new Date();
   170→
   171→    for (let i = days - 1; i >= 0; i--) {
   172→      const date = new Date(now);
   173→      date.setDate(date.getDate() - i);
   174→      date.setHours(0, 0, 0, 0);
   175→
   176→      const nextDate = new Date(date);
   177→      nextDate.setDate(nextDate.getDate() + 1);
   178→
   179→      const [total, mastered] = await Promise.all([
   180→        this.prisma.mistake.count({
   181→          where: {
   182→            userId: BigInt(userId),
   183→            subjectId,
   184→            createdAt: { lt: nextDate },
   185→          },
   186→        }),
   187→        this.prisma.mistake.count({
   188→          where: {
   189→            userId: BigInt(userId),
   190→            subjectId,
   191→            mastered: true,
   192→            masteredAt: { lt: nextDate },
   193→          },
   194→        }),
   195→      ]);
   196→
   197→      trend.push({
   198→        date: date.toISOString().split('T')[0],
   199→        masterRate: total > 0 ? Number(((mastered / total) * 100).toFixed(1)) : 0,
   200→      });
   201→    }
   202→
   203→    return trend;
   204→  }
   205→
   206→  /**
   207→   * 计算连续复习天数
   208→   */
   209→  private async calculateReviewStreak(userId: number): Promise<number> {
   210→    const reviews = await this.prisma.reviewRecord.findMany({
   211→      where: { userId: BigInt(userId) },
   212→      orderBy: { createdAt: 'desc' },
   213→      take: 365,
   214→      select: { createdAt: true },
   215→    });
   216→
   217→    if (reviews.length === 0) return 0;
   218→
   219→    let streak = 1;
   220→    const today = new Date();
   221→    today.setHours(0, 0, 0, 0);
   222→
   223→    // 检查今天是否有复习
   224→    const lastReview = new Date(reviews[0].createdAt);
   225→    lastReview.setHours(0, 0, 0, 0);
   226→
   227→    if (lastReview.getTime() < today.getTime()) {
   228→      // 今天没有复习，检查昨天
   229→      const yesterday = new Date(today);
   230→      yesterday.setDate(yesterday.getDate() - 1);
   231→
   232→      if (lastReview.getTime() < yesterday.getTime()) {
   233→        return 0; // 昨天也没有，连续天数中断
   234→      }
   235→    }
   236→
   237→    // 计算连续天数
   238→    for (let i = 1; i < reviews.length; i++) {
   239→      const currentDate = new Date(reviews[i - 1].createdAt);
   240→      currentDate.setHours(0, 0, 0, 0);
   241→
   242→      const prevDate = new Date(reviews[i].createdAt);
   243→      prevDate.setHours(0, 0, 0, 0);
   244→
   245→      const diffDays = Math.floor(
   246→        (currentDate.getTime() - prevDate.getTime()) / (1000 * 60 * 60 * 24)
   247→      );
   248→
   249→      if (diffDays === 0) continue; // 同一天
   250→      if (diffDays === 1) {
   251→        streak++;
   252→      } else {
   253→        break; // 中断
   254→      }
   255→    }
   256→
   257→    return streak;
   258→  }
   259→}
