     1→import { Injectable, NotFoundException } from '@nestjs/common';

     4→
     5→@Injectable()

     9→  /**
    10→   * 获取子账号列表
    11→   */
    12→  async findAll(parentId: number) {
    13→    const children = await this.prisma.child.findMany({
    14→      where: { parentId: BigInt(parentId), status: 1 },
    15→      orderBy: { createdAt: 'desc' },
    16→    });
    17→
    18→    // 获取每个孩子的统计信息
    19→    const result = await Promise.all(
    20→      children.map(async (child) => {
    21→        const [totalMistakes, masteredCount] = await Promise.all([
    22→          this.prisma.mistake.count({
    23→            where: { childId: BigInt(child.id) },
    24→          }),
    25→          this.prisma.mistake.count({
    26→            where: { childId: BigInt(child.id), mastered: true },
    27→          }),
    28→        ]);
    29→
    30→        return {
    31→          id: Number(child.id),
    32→          name: child.name,
    33→          grade: child.grade,
    34→          avatar: child.avatar,
    35→          birthDate: child.birthDate,
    36→          school: child.school,
    37→          stats: {
    38→            totalMistakes,
    39→            masterRate: totalMistakes > 0
    40→              ? Number(((masteredCount / totalMistakes) * 100).toFixed(1))
    41→              : 0,
    42→          },
    43→        };
    44→      }),
    45→    );
    46→
    47→    return result;
    48→  }
    49→
    50→  /**
    51→   * 获取子账号详情

    72→  /**

    87→    return {
    88→      id: Number(child.id),

    93→  }

    95→  /**
    96→   * 更新子账号
    97→   */
    98→  async update(id: number, parentId: number, dto: UpdateChildDto) {
    99→    const child = await this.prisma.child.findFirst({
   100→      where: { id: BigInt(id), parentId: BigInt(parentId) },
   101→    });
   102→
   103→    if (!child) {
   104→      throw new NotFoundException('Child not found');
   105→    }
   106→
   107→    const updated = await this.prisma.child.update({
   108→      where: { id: BigInt(id) },
   109→      data: {
   110→        ...(dto.name && { name: dto.name }),

   113→        ...(dto.birthDate && { birthDate: new Date(dto.birthDate) }),
   114→        ...(dto.school !== undefined && { school: dto.school }),
   115→      },
   116→    });
   117→
   118→    return {
   119→      id: Number(updated.id),
   120→      name: updated.name,
   121→      grade: updated.grade,

   124→  }
   125→
   126→  /**
   127→   * 删除子账号

   129→  async remove(id: number, parentId: number) {

   131→      where: { id: BigInt(id), parentId: BigInt(parentId) },
   132→    });

   156→
   157→    // 返回子账号信息，token 生成在 auth 模块处理
