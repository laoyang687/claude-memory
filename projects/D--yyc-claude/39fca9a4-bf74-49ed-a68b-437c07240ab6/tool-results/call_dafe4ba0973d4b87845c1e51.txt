     1→import { Injectable } from '@nestjs/common';
     2→import { PrismaService } from '@/common/services/prisma.service';
     3→import { ReviewResult } from '@prisma/client';
     4→
     5→@Injectable()
     6→export class ReviewService {
     7→  constructor(private prisma: PrismaService) {}
     8→
     9→  /**
    10→   * 获取今日待复习
    11→   */
    12→  async getTodayReviews(userId: number) {
    13→    const now = new Date();
    14→    const startOfDay = new Date(now);
    15→    startOfDay.setHours(0, 0, 0, 0);
    16→    const endOfDay = new Date(startOfDay);
    17→    endOfDay.setDate(endOfDay.getDate() + 1);
    18→
    19→    const [todayList, overdueList, completedToday] = await Promise.all([
    20→      // 今日应复习
    21→      this.prisma.mistake.findMany({
    22→        where: {
    23→          userId: BigInt(userId),
    24→          mastered: false,
    25→          nextReviewAt: { lte: endOfDay },
    26→        },
    27→        include: { subject: true },
    28→        take: 50,
    29→        orderBy: { nextReviewAt: 'asc' },
    30→      }),
    31→      // 超期未复习
    32→      this.prisma.mistake.findMany({
    33→        where: {
    34→          userId: BigInt(userId),
    35→          mastered: false,
    36→          nextReviewAt: { lt: startOfDay },
    37→        },
    38→        include: { subject: true },
    39→        orderBy: { nextReviewAt: 'asc' },
    40→      }),
    41→      // 今日已完成
    42→      this.prisma.reviewRecord.count({
    43→        where: {
    44→          userId: BigInt(userId),
    45→          createdAt: { gte: startOfDay, lt: endOfDay },
    46→        },
    47→      }),
    48→    ]);
    49→
    50→    const todayCount = todayList.length;
    51→    const overdueCount = overdueList.length;
    52→
    53→    return {
    54→      todayCount,
    55→      overdueCount,
    56→      completedToday,
    57→      list: todayList.map((item) => ({
    58→        id: Number(item.id),
    59→        questionText: item.questionText,
    60→        questionImages: item.questionImages ? JSON.parse(item.questionImages) : [],
    61→        subject: {
    62→          id: item.subject.id,
    63→          name: item.subject.name,
    64→          icon: item.subject.icon,
    65→          color: item.subject.color,
    66→        },
    67→        isOverdue: item.nextReviewAt < startOfDay,
    68→        reviewCount: item.reviewCount,
    69→        lastReviewAt: item.lastReviewAt,
    70→        nextReviewAt: item.nextReviewAt,
    71→      })),
    72→    };
    73→  }
    74→
    75→  /**
    76→   * 提交复习结果
    77→   */
    78→  async submitReview(
    79→    userId: number,
    80→    mistakeId: number,
    81→    result: ReviewResult,
    82→    duration?: number,
    83→    notes?: string,
    84→  ) {
    85→    const mistake = await this.prisma.mistake.findFirst({
    86→      where: {
    87→        id: BigInt(mistakeId),
    88→        userId: BigInt(userId),
    89→      },
    90→    });
    91→
    92→    if (!mistake) {
    93→      throw new Error('Mistake not found');
    94→    }
    95→
    96→    // 创建复习记录
    97→    await this.prisma.reviewRecord.create({
    98→      data: {
    99→        mistakeId: BigInt(mistakeId),
   100→        userId: BigInt(userId),
   101→        result,
   102→        duration,
   103→        notes,
   104→      },
   105→    });
   106→
   107→    // 更新错题状态
   108→    const newReviewCount = mistake.reviewCount + 1;
   109→    const isCorrect = result === 'correct';
   110→
   111→    // 计算下次复习时间
   112→    const nextReviewAt = this.calculateNextReview(newReviewCount, result);
   113→
   114→    await this.prisma.mistake.update({
   115→      where: { id: BigInt(mistakeId) },
   116→      data: {
   117→        reviewCount: newReviewCount,
   118→        lastReviewAt: new Date(),
   119→        nextReviewAt,
   120→        // 连续答对3次标记为已掌握
   121→        mastered: isCorrect && newReviewCount >= 3,
   122→      },
   123→    });
   124→
   125→    return {
   126→      nextReviewAt,
   127→      reviewCount: newReviewCount,
   128→      mastered: isCorrect && newReviewCount >= 3,
   129→    };
   130→  }
   131→
   132→  /**
   133→   * 获取复习统计
   134→   */
   135→  async getReviewStats(userId: number, startDate?: Date, endDate?: Date) {
   136→    const where: any = { userId: BigInt(userId) };
   137→
   138→    if (startDate || endDate) {
   139→      where.createdAt = {};
   140→      if (startDate) where.createdAt.gte = startDate;
   141→      if (endDate) where.createdAt.lte = endDate;
   142→    }
   143→
   144→    const [totalReviews, correctCount, avgDuration] = await Promise.all([
   145→      this.prisma.reviewRecord.count({ where }),
   146→      this.prisma.reviewRecord.count({
   147→        where: { ...where, result: 'correct' },
   148→      }),
   149→      // 平均用时需要聚合查询
   150→      this.prisma.reviewRecord.aggregate({
   151→        where: { ...where, duration: { not: null } },
   152→        _avg: { duration: true },
   153→      }),
   154→    ]);
   155→
   156→    return {
   157→      totalReviews,
   158→      correctCount,
   159→      correctRate: totalReviews > 0
   160→        ? Number(((correctCount / totalReviews) * 100).toFixed(1))
   161→        : 0,
   162→      avgDuration: avgDuration._avg.duration || 0,
   163→    };
   164→  }
   165→
   166→  /**
   167→   * 计算下次复习时间（艾宾浩斯遗忘曲线）
   168→   */
   169→  private calculateNextReview(reviewCount: number, result: ReviewResult): Date {
   170→    const intervals = [1, 2, 4, 7, 15, 30, 60, 90]; // 天数
   171→
   172→    let idx: number;
   173→    if (result === 'wrong') {
   174→      idx = 0; // 答错重置
   175→    } else if (result === 'partial') {
   176→      idx = Math.max(0, reviewCount - 1); // 部分正确保持当前间隔
   177→    } else {
   178→      idx = Math.min(reviewCount, intervals.length - 1); // 全对推进
   179→    }
   180→
   181→    const nextDate = new Date();
   182→    nextDate.setDate(nextDate.getDate() + intervals[idx]);
   183→
   184→    return nextDate;
   185→  }
   186→}
