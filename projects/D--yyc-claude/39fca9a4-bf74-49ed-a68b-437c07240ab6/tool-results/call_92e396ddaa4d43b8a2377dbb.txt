    23→model User {
    24→  id        BigInt   @id @default(autoincrement())

    26→  unionid   String?  @db.VarChar(64)

    29→  role      UserRole @default(student)
    30→  phone     String?  @db.VarChar(20)
    31→  grade     String?  @db.VarChar(20)
    32→  status    Int      @default(1) @db.TinyInt
    33→  createdAt DateTime @default(now()) @map("created_at")
    34→  updatedAt DateTime @updatedAt @map("updated_at")
    35→
    36→  // 关系
    37→  children      Child[]         @relation("ParentChildren")
    38→  mistakes      Mistake[]
    39→  reviewRecords ReviewRecord[]
    40→  practiceRecords PracticeRecord[]
    41→
    42→  @@index([openid])
    43→  @@index([grade])
    44→  @@index([role])

    54→  birthDate DateTime? @map("birth_date")
    55→  school    String?  @db.VarChar(100)
    56→  status    Int      @default(1) @db.TinyInt
    57→  createdAt DateTime @default(now()) @map("created_at")
    58→  updatedAt DateTime @updatedAt @map("updated_at")
    59→
    60→  // 关系
    61→  parent         User            @relation("ParentChildren", fields: [parentId], references: [id], onDelete: Cascade)

    65→  @@index([parentId])

   114→enum SourceType {
   115→  exam
   116→  homework
   117→  practice

   120→enum QuestionType {
   121→  choice
   122→  fill

   127→enum WrongReason {
   128→  concept    // 概念错误
   129→  calc       // 计算错误

   131→  unknown    // 未知

   134→model Mistake {
   135→  id             BigInt      @id @default(autoincrement())
   136→  userId         BigInt      @map("user_id")
   137→  childId        BigInt?     @map("child_id")
   138→  subjectId      Int         @map("subject_id")
   139→  knowledgeIds   String?     @map("knowledge_ids") @db.VarChar(500)
   140→  sourceType     SourceType  @default(exam) @map("source_type")
   141→
   142→  // 题目信息
   143→  questionText   String      @map("question_text") @db.Text
   144→  questionImages String?     @map("question_images") @db.Text // JSON string
   145→  questionType   QuestionType? @map("question_type")
   146→
   147→  // 错误信息
   148→  wrongAnswer    String?     @map("wrong_answer") @db.Text
   149→  wrongReason    WrongReason? @map("wrong_reason")
   150→  aiAnalysis     String?     @map("ai_analysis") @db.Text
   151→
   152→  // 正确信息
   153→  correctAnswer  String?     @map("correct_answer") @db.Text
   154→  explanation    String?     @db.Text
   155→  solutionImages String?     @map("solution_images") @db.Text // JSON string
   156→
   157→  // 标签与分类
   158→  tags           String?     @db.VarChar(500)
   159→  difficulty     Int         @default(3) @db.TinyInt
   160→
   161→  // 状态
   162→  mastered       Boolean     @default(false) @map("mastered") @db.TinyInt(1)
   163→  reviewCount    Int         @default(0) @map("review_count")
   164→  lastReviewAt   DateTime?   @map("last_review_at")
   165→  nextReviewAt   DateTime?   @map("next_review_at")
   166→
   167→  // 时间
   168→  createdAt      DateTime    @default(now()) @map("created_at")
   169→  updatedAt      DateTime    @updatedAt @map("updated_at")
   170→
   171→  // 关系
   172→  user         User           @relation(fields: [userId], references: [id], onDelete: Cascade)
   173→  child        Child?         @relation(fields: [childId], references: [id])
   174→  subject      Subject        @relation(fields: [subjectId], references: [id])
   175→  reviewRecords ReviewRecord[]
   176→
   177→  @@index([userId, subjectId])
   178→  @@index([childId])
   179→  @@index([userId, nextReviewAt])
   180→  @@index([userId, mastered])
   181→  @@index([createdAt])
   182→  @@map("mistakes")
   183→}
