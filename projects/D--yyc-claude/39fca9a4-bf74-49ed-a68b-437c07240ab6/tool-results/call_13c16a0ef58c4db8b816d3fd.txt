     9→  /**

    12→  async findAll(userId?: number) {

    32→          color: tag.color,

    34→        };

    36→    );
    37→
    38→    return result;
    39→  }
    40→
    41→  /**
    42→   * 创建标签

    44→  async create(userId: number, dto: CreateTagDto) {

    52→    });
    53→
    54→    return {
    55→      id: Number(tag.id),
    56→      name: tag.name,

    58→    };
    59→  }
    60→
    61→  /**
    62→   * 删除标签

    64→  async remove(id: number, userId: number) {

    66→      where: {
    67→        id: BigInt(id),
    68→        OR: [{ userId: BigInt(userId) }, { userId: null }],
    69→      },
    70→    });
    71→
    72→    if (!tag) {
    73→      throw new Error('Tag not found');
    74→    }
    75→
    76→    await this.prisma.tag.delete({

    81→  }
